List employee name and name of managers.
SELECT e.ENAME AS Employee,
    ->        m.ENAME AS Manager
    -> FROM emp e
    -> LEFT JOIN emp m
    -> ON e.MGR = m.EMPNO;

Display details of all departments and the corresponding employees
SELECT d.DEPTNO,
    ->        d.DNAME,
    ->        d.LOC,
    ->        e.EMPNO,
    ->        e.ENAME,
    ->        e.JOB,
    ->        e.SAL
    -> FROM dept d
    -> LEFT JOIN emp e
    -> ON d.DEPTNO = e.DEPTNO;
 
Display all customer names with their order details (order number and ship date) for the orders shipped in 1987.
select c.name,o.ordid,o.shipdate from ord o join customer c on o.custid= c.custid where year(o.shipdate)=1987;

Display salesman names and names of customers they are dealing with.
select s.sname,c.cname from customers c join salespeople s on s.snum = c.snum;
